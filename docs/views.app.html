    activateFormAddView: ->
      @activeSubview = @subviews['formAddView']
      if not @activeSubview.initialized
        @activeSubview = new @activeSubview()
        @activeSubview.parent = @
        @subviews['formAddView'] = @activeSubview
      @activeSubview.setElement '#appview'
      @renderSubview()
