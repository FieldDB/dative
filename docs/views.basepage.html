    Page Views contain a header div (#old-page-header) and a body div (#old-page-body). matchHeights keeps the body height constant relative to its parentâ€™s height, even when window resizes.

          matchHeights: ->
        pageBody = $('#old-page-body')[0]
        parent = $(pageBody).parent()
        pageHeader = $('#old-page-header')
        margin = parseInt $(pageBody).css('margin')
        hasVerticalScrollBar = @hasVerticalScrollBar
        $(pageBody).css 'height',
          parent.innerHeight() - pageHeader.outerHeight() - (margin * 2)
        if hasVerticalScrollBar pageBody
          $(pageBody).css 'padding-right', 10
        $(window).resize ->
          $(pageBody).css 'height',
            parent.innerHeight() - pageHeader.outerHeight() - (margin * 2)
          if hasVerticalScrollBar pageBody
            $(pageBody).css 'padding-right', 10
